Import-Module ActiveDirectory
Write-Host "`n--- Initializing Enterprise AD Health Assessment ---" -ForegroundColor Cyan

$Report = @()
$CSVPath = "$HOME\Desktop\AD_Infrastructure_Health_Report.csv"

# 1. Audit Domain Controllers & FSMO Roles
Write-Host "[AUDIT] Fetching Domain Controller & FSMO Topology..." -ForegroundColor Gray
try {
    $Domain = Get-ADDomain "lab.local"
    $Forest = Get-ADForest "lab.local"
    
    Write-Host "[SUCCESS] PDC Emulator: $($Domain.PDCEmulator)" -ForegroundColor Green
    Write-Host "[SUCCESS] Schema Master: $($Forest.SchemaMaster)" -ForegroundColor Green
} catch {
    Write-Host "[ERROR] Failed to query domain topology: $_" -ForegroundColor Red
}

# 2. Audit Sites and Services Topology
Write-Host "`n[AUDIT] Extracting AD Sites and Subnet Mappings..." -ForegroundColor Gray
$Sites = [System.DirectoryServices.ActiveDirectory.ActiveDirectorySite]::GetComputerSite()
$AllSites = (Get-ADReplicationSite -Filter *).Name

foreach ($Site in $AllSites) {
    Write-Host "[INFO] Detected Configured AD Site: $Site" -ForegroundColor Yellow
}

# 3. Check Replication Status (Crucial for Multi-Forest/Multi-Site roles)
Write-Host "`n[AUDIT] Analyzing Metadata Replication Status..." -ForegroundColor Gray
try {
    # Simulating replication metadata check for the local DC
    $ReplStatus = Get-ADReplicationPartnerMetadata -Target "localhost" -ErrorAction Stop
    Write-Host "[SUCCESS] Replication inbound links are healthy." -ForegroundColor Green
} catch {
    # In a single DC lab environment, this will gracefully catch or show baseline metadata
    Write-Host "[WARN] Single DC environment detected or replication link status unreadable. Skipping sync audit." -ForegroundColor Yellow
}

# 4. Compile a Health Log Summary for Splunk Ingestion
Write-Host "`n[COMPLIANCE] Packaging metrics into structured log export..." -ForegroundColor Gray

$HealthObject = [PSCustomObject]@{
    Timestamp      = (Get-Date -Format "yyyy-MM-dd HH:mm:ss")
    DomainName     = "lab.local"
    PDCEmulator    = $Domain.PDCEmulator
    SchemaMaster   = $Forest.SchemaMaster
    TotalConfigSites = ($AllSites).Count
    Status         = "OPERATIONAL"
}

$Report += $HealthObject
$Report | Export-Csv -Path $CSVPath -NoTypeInformation
Write-Host "[SUCCESS] System health log compiled and exported to: $CSVPath" -ForegroundColor Cyan
Write-Host "--- Health Assessment Cycle Complete ---`n" -ForegroundColor Cyan
